class MeshyPipeline:
    def process_concept(self, image_path):
        # Upload to Meshy AI
        task_id = self.meshy_api.create_task(image_path)
        
        # Poll for completion

function exportMarkersToJSON(comp) {
    var markers = [];
    var markerProp = comp.markerProperty;
    
    for (var i = 1; i <= markerProp.numKeys; i++) {
        markers.push({
            time: markerProp.keyTime(i),
            comment: markerProp.keyValue(i).comment,
            duration: markerProp.keyValue(i).duration
        });
    }
    
    return JSON.stringify(markers, null, 2);
}

def get_artist_workload(self, artist_id):
    tasks = self.sg.find("Task", [
        ["task_assignees", "is", {"type": "HumanUser", "id": artist_id}],
        ["sg_status_list", "not_in", ["fin", "omt"]]
    ], ["content", "sg_status_list", "due_date"])
    
    return {
        "total": len(tasks),
        "in_progress": sum(1 for t in tasks if t["sg_status_list"] == "ip"),
        "tasks": tasks
    }
