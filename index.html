  • Strong modifier stack
  • Arch-viz focused

  • Cross-platform (Win/Mac/Linux)
  • Python & MEL scripting
  • Native USD integration
  • Industry-standard rigging
  • VFX/Animation focused

import maya.cmds as cmds
from pathlib import Path

def migrate_max_asset(fbx_path: str, output_dir: str) -> dict:
    """Import FBX from Max export and restructure for Maya pipeline"""
    
    # Import FBX with specific settings
    cmds.file(fbx_path, i=True, type="FBX",
              options="fbx;importMode=merge")
    
    # Fix Max-specific naming conventions
    for node in cmds.ls(type="transform"):
        if node.startswith("Box") or node.startswith("Cylinder"):
            new_name = convert_max_naming(node)
            cmds.rename(node, new_name)
    
    # Convert V-Ray materials to Arnold
    vray_mats = cmds.ls(type="VRayMtl")
    for mat in vray_mats:
        convert_vray_to_arnold(mat)
    
    # Rebuild hierarchy to Maya standards
    create_maya_hierarchy()
    
    return {"status": "success", "nodes": cmds.ls(dag=True)}
